示例1：Tab标签页刷新脚本

Tab组件有一个 onChange 事件，脚本写在里边。

function onTabClick(tab) {
    if (tab === 0) { // 点击第1个tab页签
        refreshTab(this, "page_widget_list_d6oo6l", "table_hwhgt");
    } else if (tab === 1) { // 点击第2个tab页签
        refreshTab(this, "page_widget_list_21rba9", "table_6uc5m2");
    }else {
        refreshTab(this, "page_widget_list_8nv8nq", "table_6cbktu");
    }
}

/**
 * pageId 在页面构建中，找到对应的页面，打开页面设计器，选中对应标签下的内容，点击高级，查看唯一标识
 * tableId 在部件设计中，找到对应的列表，打开列表设计器，点击高级，查看唯一标识
 */
function refreshTab(context, pageId, tableId){
    // 找到页面组件
    var pageComp = context.findElement(pageId);
    if (pageComp && pageComp.dataEngine && pageComp.dataEngine.state) {
        // 找到页面下的表格组件
        var tableComp = pageComp.dataEngine.state.getElement(tableId); 
        // 调用表格的刷新方法
        tableComp && tableComp.onRefresh();
    }
}
